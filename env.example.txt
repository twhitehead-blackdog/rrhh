# ============================================
# Peopletrak - Variables de Entorno de Ejemplo
# ============================================
# IMPORTANTE: Todas las variables deben comenzar con ENV_
# El plugin env-var-plugin.js solo inyecta variables con este prefijo
#
# Copia este archivo como .env y completa con tus valores reales
# ============================================

# ============================================
# Auth0 Configuration
# ============================================
# Obtén estos valores de: https://manage.auth0.com/
# 1. Ve a Applications > Tu Aplicación
# 2. Domain: Lo encuentras en la página principal
# 3. Client ID: Lo encuentras en la página de Settings
# 4. Audience: Lo encuentras en APIs > Tu API

ENV_AUTH0_DOMAIN=tu-dominio.auth0.com
ENV_AUTH0_CLIENT_ID=tu-client-id-aqui
ENV_AUTH0_AUDIENCE=https://tu-api.com

# ============================================
# Supabase Configuration
# ============================================
# Obtén estos valores de: https://app.supabase.com/
# 1. Ve a tu proyecto
# 2. Settings > API
# 3. URL: Project URL
# 4. API Key: anon/public key

ENV_SUPABASE_URL=https://tu-proyecto.supabase.co
ENV_SUPABASE_API_KEY=tu-api-key-publica-aqui

# ============================================
# Application Configuration
# ============================================
# URL donde se ejecutará la aplicación localmente

ENV_APP_URL=http://localhost:4200

# ============================================
# Notas
# ============================================
# - En desarrollo, usa http://localhost:4200
# - En producción, usa tu dominio real
# - No subas este archivo con valores reales a Git
# - El archivo .env está en .gitignore por seguridad
